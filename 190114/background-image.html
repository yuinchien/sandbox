3

Changing background color

<model-viewer id="toggle-color" alt="A 3D model of a shishkebab" src="assets/shishkebab.glb" controls auto-rotate>
  </model-viewer>
  <script>
      const colors = ['#4885ed', '#3cba54', '#f4c20d', '#db3236'];
      let i = 0;
      setInterval(() =>
          $('#toggle-color').setAttribute('background-color', colors[i++ % 4]), 2000);
  </script>
